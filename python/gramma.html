classmethod & staticmethod ​

python
class Test:
    def __init__(self, name):
        pass
        self.name = name

    def run(self):
        print("run")
        self.dump()
        self.sleep()

    @classmethod
    def dump(cls):
        print("dump")

    @staticmethod
    def sleep():
        print("sleep")

    @property
    def name(self):
        return self.__name

    @name.setter
    def name(self, value):
        self.__name = value


self = Test('start')
self.run()
Test.dump()

self.sleep()
Test.sleep()

print(self.name)
self.name = 'after sleep'
print(self.name)
